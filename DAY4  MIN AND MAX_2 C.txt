//Min_Max
#include <stdio.h>

#define SIZE 10

int main() {
    int arr[SIZE] = {64, 84, 20, 36, 12, 90, 72, 10, 45, 56};
    int min = arr[0]; 
    int max = arr[0]; 
    int i;

    // Find the minimum and maximum
    for (i = 1; i < SIZE; i++) {
        if (arr[i] < min) {
            min = arr[i];
        }
        if (arr[i] > max) {
            max = arr[i];
        }
    }

    printf("Minimum: %d\n", min);
    printf("Maximum: %d\n", max);

    return 0;
}
/*OUTPUT
Minimum: 10
Maximum: 90*/
